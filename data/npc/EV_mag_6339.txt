#여행의 문을 관리하는 ＮＰＣ. 워프시키기, 프래그 세트하기의 기능이 있다.

block Item 18440 >= 1

	EndWindow 14660,OK

############## 여행의 문 프래그（100）가 서 있지 않다 ##############
block NowEvent 100 == 0
#\n\n　제노바에 어서 오세요. 여기는 판부르크와 제노바를 잇는 게이트입니다. 이후부터 이 게이트를 통해서 판부르크와 왕래할 수가 있습니다. 많이 이용해주십시오.
	NowEvent 100 = 1
	Endwindow 14665,OK

############## 그룹을 이루고 있다 ##############
block Group != 1
#\n\n　게이트는 한 분씩 밖에 이용할 수 없습니다.
	Endwindow 14601,OK

############## 여행의 문 프래그가 서 있고, 그룹을 이루고 있지 않다 ##############
block

	Window 14666,YES,NO

############## 프리패스사용 ##############
	Window_YES Group == 1 && Item 18660 >= 1

	Window 15427,OK
