### 무기수리직인이 아닌 경우 ###

	block Job >= 0 && Job < 370
	#곤란하게 되었군.\n무기수리에 능숙한 분 안 계십니까.
	EndWindow 10844

	block Job >= 380 && Job < 470
	#곤란하게 되었군.\n무기수리에 능숙한 분 안 계십니까.
	EndWindow 10844


### 무기수리직인이고, 19122를 가지고 있는 경우 ###

block Job >= 370 && Job < 380 && Item19122 >= 1
	#그것은 내 「츠라누키마루」가 아닌가!\n고쳐주신 거로군요.\n고맙소!
	Window 10843,OK

	 # --- OK의 경우 --- # 
	Window_OK
	KillItem19122
	#얼마 되지 않지만 10GOLD입니다.\n내 감사의 마음이오.
	EndWindow 10842


### 무기수리직인이고, 19121을 가지고 있는 경우 ###

block Job >= 370 && Job < 380 && Item19121 >= 1
	#가능한 한 빨리 고쳐주었으면 좋겠소.\n좀 있으면 양아버지가 돌아오실 거요.
	EndWindow 10841


### 무기수리직인이고, 19121을 가지고 있지 않은 경우 ###

block Job >= 370 && Job < 380 && Item19121 == 0
	#아아, 큰일이군 큰일이야.\n어라, 그쪽은 무기수리직인이신가?\n\n실은 양아버지로부터 받은 소중한 「츠라누키마루」를\n부러뜨려 버렸다네. 제발 이것을 수선해 주시오.
	Window 10840,YES,NO

	 # --- YES의 경우 --- # 
	Window_YES
	Additem19121
	#수선이 끝나면, 나에게 가지고\n와 주시게. 부탁하네.
	EndWindow 10839


	 # --- NO의 경우 --- # 
	Window_NO
	#아아, 빌보가 화를 낼텐데. 큰일이군 큰일이야.
	EndWindow 10838
